plot1 = instacart %>%
  select(aisle, department)%>% 
  group_by(department) %>% 
   count(aisle) %>% 
  arrange(desc(n)) %>% 
  filter(n > 20000) %>% 
  mutate(
    aisle = fct_reorder( aisle, n),
    text_label = str_c("department", department,"aisle", aisle, "count", n)) %>% 
  plot_ly(x = ~aisle, y = ~n, color = ~department, type = "bar")
